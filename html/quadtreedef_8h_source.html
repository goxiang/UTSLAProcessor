166template<class T>
167bool MQuadTree::updatePathInfo(const MPtInfoPtr &ptPtr, const QVector<T> &srcVec,
168 QVector<T> &desVec)
169{
170 _curPtPtr = ptPtr;
171 if (nullptr == _curPtPtr) return false;
172 _curBroPtr = _curPtPtr->_bro.toStrongRef();
173 if (nullptr == _curBroPtr) return false;
174
175 removePt(_curPtPtr);
176
177 if (ptPtr->_index >= srcVec.size()) return false;
178
179 if (ptPtr->_startType)
180 {
181 for (int index = _curBroPtr->_index; index <= _curPtPtr->_index; ++ index)
182 {
183 auto path = srcVec.at(index);
184 // ReversePath(path);
185 desVec.push_back(path);
186 }
187 }
188 else
189 {
190 for (int index = _curPtPtr->_index; index <= _curBroPtr->_index; ++ index)
191 {
192 desVec.push_back(srcVec.at(index));
193 }
194 }
195 return true;
196}
